                  let strNew = []
                  str = str.split(' ')

                    if (item.match( (/[^A-Za-z0-9\s]/g))){
                      strNew.push(item)
                        }
                        else {
                            strNew.push(item.slice(1)

                        }
                    })
                    return strNew.join(' ')
                }
